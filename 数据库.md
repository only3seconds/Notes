# 数据库

## SQL

### 1. 创建数据库

	CREATE DATABASE test;
	
### 2. 使用数据库

	USE test;
	
### 3. 创建表
	
	CREATE TABLE Orders
	(
	Id_O int NOT NULL,
	OrderNo int NOT NULL,
	Id_P int,
	PRIMARY KEY (Id_O),
	CONSTRAINT fk_PerOrders FOREIGN KEY (Id_P) REFERENCES Persons(Id_P)
	)
